About This Book

This comprehensive guide presents probability theory, statistics, and data analysis from a Bayesian perspective for physicists and engineers. Key topics include:

Part I: Introduction

Fundamental concepts of probability including classical, statistical, and Bayesian interpretations, combinatorics, random walks, and limit theorems.

Part II: Assigning Probabilities

Methods for constructing prior probabilities including transformation invariance, maximum entropy principles, and smoothness constraints.

Part III: Parameter Estimation

Both Bayesian and frequentist approaches to estimating unknown parameters, including the Cramer-Rao inequality.

Part IV: Testing Hypotheses

Bayesian hypothesis testing, frequentist methods, sampling distributions, and comparisons between the two approaches.

Part V: Real-World Applications

Practical applications including regression, inconsistent data handling, signal separation, change point detection, function estimation, and more.

Part VI: Probabilistic Numerical Techniques

Advanced computational methods including numerical integration, Monte Carlo methods, and nested sampling.

Throughout this book, you’ll find code examples in R and Python. Here’s a quick example of Bayesian inference:

Example 1: Simple Bayesian Inference in R

# Install required package if needed
# install.packages("tidyverse")

library(ggplot2)

# Define prior, likelihood, and posterior
prior <- function(theta) dbeta(theta, 1, 1)  # Uniform prior
likelihood <- function(theta, data) prod(dbinom(data, size=1, prob=theta))
posterior <- function(theta, data) likelihood(theta, data) * prior(theta)

# Generate some example data (10 coin flips, 7 heads)
data <- c(1, 1, 0, 1, 1, 1, 1, 0, 1, 1)

# Compute posterior on a grid
theta_grid <- seq(0, 1, length.out = 100)
posteriors <- sapply(theta_grid, function(t) posterior(t, data))
posteriors <- posteriors / sum(posteriors)

# Visualize
ggplot(data.frame(theta = theta_grid, posterior = posteriors), 
       aes(x = theta, y = posterior)) +
  geom_line(color = "steelblue", size = 1) +
  geom_area(fill = "steelblue", alpha = 0.3) +
  theme_minimal() +
  labs(title = "Posterior Distribution of θ (Coin Bias)",
       x = "θ (probability of heads)",
       y = "Posterior Probability Density")

Example 2: Monte Carlo Integration in Python

import numpy as np
import matplotlib.pyplot as plt

# Define the function to integrate: f(x) = exp(-x^2)
def f(x):
    return np.exp(-x**2)

# Monte Carlo integration
def monte_carlo_integration(func, a, b, n_samples=10000):
    """
    Estimate integral of func from a to b using Monte Carlo method
    """
    x = np.random.uniform(a, b, n_samples)
    y = func(x)
    integral_estimate = (b - a) * np.mean(y)
    return integral_estimate, x, y

# Estimate integral from -3 to 3
a, b = -3, 3
estimate, x_samples, y_samples = monte_carlo_integration(f, a, b, n_samples=5000)
true_value = np.sqrt(np.pi) * (2/np.sqrt(np.pi)) * 0.997  # Approximate

print(f"Monte Carlo estimate: {estimate:.4f}")
print(f"Approximate true value: {true_value:.4f}")

# Visualize
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 4))

# Plot 1: Function and samples
x_plot = np.linspace(a, b, 200)
ax1.plot(x_plot, f(x_plot), 'b-', linewidth=2, label='f(x) = exp(-x²)')
ax1.scatter(x_samples, y_samples, alpha=0.3, s=10, label='MC samples')
ax1.fill_between(x_plot, 0, f(x_plot), alpha=0.2)
ax1.set_xlabel('x')
ax1.set_ylabel('f(x)')
ax1.set_title('Monte Carlo Integration')
ax1.legend()
ax1.grid(True, alpha=0.3)

# Plot 2: Convergence
n_range = np.logspace(1, 4, 20, dtype=int)
estimates = []
for n in n_range:
    est, _, _ = monte_carlo_integration(f, a, b, n_samples=n)
    estimates.append(est)

ax2.loglog(n_range, estimates, 'o-', label='MC estimates')
ax2.axhline(y=true_value, color='r', linestyle='--', label='True value')
ax2.set_xlabel('Number of samples')
ax2.set_ylabel('Integral estimate')
ax2.set_title('Convergence of Monte Carlo Estimator')
ax2.legend()
ax2.grid(True, alpha=0.3, which='both')

plt.tight_layout()
plt.show()

Example 3: Markov Chain Monte Carlo in R

# Simple Metropolis-Hastings sampler
metropolis_hastings <- function(log_target, n_iterations = 10000, 
                               proposal_sd = 1, initial_value = 0) {
  
  samples <- numeric(n_iterations)
  current <- initial_value
  n_accepted <- 0
  
  for (i in 1:n_iterations) {
    # Propose new value
    proposed <- current + rnorm(1, 0, proposal_sd)
    
    # Compute acceptance ratio (on log scale for numerical stability)
    log_ratio <- log_target(proposed) - log_target(current)
    
    # Accept/reject
    if (log(runif(1)) < log_ratio) {
      current <- proposed
      n_accepted <- n_accepted + 1
    }
    
    samples[i] <- current
  }
  
  acceptance_rate <- n_accepted / n_iterations
  cat(sprintf("Acceptance rate: %.2f%%\n", 100 * acceptance_rate))
  
  return(list(samples = samples, acceptance_rate = acceptance_rate))
}

# Example: Sample from a mixture of two normal distributions
log_target <- function(x) {
  log(0.3 * dnorm(x, mean = -3, sd = 1) + 
      0.7 * dnorm(x, mean = 2, sd = 1))
}

# Run sampler
result <- metropolis_hastings(log_target, n_iterations = 5000, proposal_sd = 1.5)

# Plot results
par(mfrow = c(2, 2))
plot(result$samples, type = 'l', main = 'Trace plot', xlab = 'Iteration', ylab = 'x')
hist(result$samples, breaks = 50, main = 'Posterior histogram', prob = TRUE)
x_range <- seq(-6, 6, 0.1)
lines(x_range, 0.3 * dnorm(x_range, -3, 1) + 0.7 * dnorm(x_range, 2, 1), 
      col = 'red', lwd = 2)
acf(result$samples, main = 'Autocorrelation')
