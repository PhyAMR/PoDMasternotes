22.1 VR traffic prediction

In this notebook, we will explore the prediction of Virtual Reality (VR) traffic. The data come from the paper:

Lecci, Mattia, et al. “An open framework for analyzing and modeling XR network traffic.” IEEE Access 9 (2021): 129782-129795.

The VR game Virus Popper was instantiated on a computer through the RiftCat application: the user could then see the virtual content on their phone, which was strapped to their head with a Cardboard viewer. The file virus_popper.csv contains three columns from the traffic capture:

idx    | frame size (B) |  time (s)
0      | 38424          |  0.0
1      | 39801          |  0.01944
...

The game was run at 60 frames per second, with a target rate of 30 Mb/s. The task is then to predict the size of the next frame, given the past N frames. This was explored in another paper:

Chiariotti, Federico, et al. “Temporal characterization and prediction of vr traffic: A network slicing use case.” IEEE Transactions on Mobile Computing 23.5 (2023): 3890-3908.

# Least squares solution
def least_squares(X_matrix: np.ndarray, labels: np.ndarray) -> None:
    # Create A matrix by multiplying X by itself
    A = np.transpose(X_matrix).dot(X_matrix)
    # Create b vector by multiplying X by the labels Y
    b = np.transpose(X_matrix).dot(labels)
    # Solve the linear system
    coeff = np.linalg.inv(A).dot(b)
    return coeff

def evaluate_model(x, y, coeff):
    # Linear prediction
    y_pred = x.dot(coeff)
    mse_vector = (y - y_pred) ** 2
    return np.sum(mse_vector) / len(mse_vector)

Solution (vectorized solution).

Show code 
# Vectorised normal-equations solver.
# Changes vs. the reference solution:
#   * use the @ operator instead of np.transpose(...).dot(...)
#   * solve the linear system directly with np.linalg.solve, which is
#     numerically more stable and faster than forming the explicit inverse
#     (A^{-1} b is replaced by solve(A, b))
#   * the MSE is a single call to np.mean on the squared residual vector
def least_squares(X_matrix: np.ndarray, labels: np.ndarray) -> None:
    # Normal equations:  (Phi.T @ Phi) w = Phi.T @ y
    A = X_matrix.T @ X_matrix
    b = X_matrix.T @ labels
    coeff = np.linalg.solve(A, b)
    return coeff

def evaluate_model(x, y, coeff):
    # Vectorised residual + mean squared error
    residual = y - x @ coeff
    return np.mean(residual ** 2)
Show code 
# Run the LS training and test it on the test data

mse = evaluate_model(X_test, Y_test, trained_model)
print('Model coefficients:', trained_model)
print('Root MSE:', np.sqrt(mse) * avg_size)

22.5 Least Squares with Tikhonov regularization

Perform K-fold cross validation with \(\lambda\in\{0, 0.1, 1, 10\}\)

Run the LS algorithm with regularization. Perform K-fold cross-validation.

Solution (solution).

Show code 
# Least squares solution
def regularized_least_squares(X_matrix: np.ndarray, labels: np.ndarray, lambda_par: np.ndarray) -> None:
    # Create A and b as in unregularized LS
    A = np.transpose(X_matrix).dot(X_matrix)
    b = np.transpose(X_matrix).dot(labels)
    # Add the regularization part in the inverse
    coeff = np.linalg.inv(A + lambda_par * np.eye(np.shape(A)[0])).dot(b)
    return coeff

def K_fold(X_training: np.ndarray, Y_training: np.ndarray, lambda_vec: np.ndarray, K: np.ndarray) -> None:
    perm = np.arange(0, np.shape(X_training)[0], 1)
    rnd.shuffle(perm)
    X_shuffled = np.asarray(X_training[perm, :])
    Y_shuffled = np.asarray(Y_training[perm])
    best = -1
    best_perf = 1e20
    results = []
    models = []
    max_idx = np.shape(Y_training)[0]
    # Divide training set in K folds
    fold_points = int(np.floor(max_idx / K))
    for lambda_idx in range(len(lambda_vec)):
        lambda_perf = 0
        for test in range(K):
            x_train_fold = np.concatenate([X_shuffled[: fold_points * test, :], X_shuffled[fold_points * (test + 1) + 1 :, :]])
            y_train_fold = np.concatenate([Y_shuffled[: fold_points * test], Y_shuffled[fold_points * (test + 1) + 1 :]])
            x_validate_fold = X_shuffled[fold_points * test : fold_points * (test + 1) + 1, :]
            y_validate_fold = Y_shuffled[fold_points * test : fold_points * (test + 1) + 1]
            k_model = regularized_least_squares(x_train_fold, y_train_fold, lambda_par[lambda_idx])
            lambda_perf += evaluate_model(x_validate_fold, y_validate_fold, k_model)
        lambda_perf /= K
        results.append(lambda_perf)
        models.append(regularized_least_squares(X_shuffled, Y_shuffled, lambda_par[lambda_idx]))
        if (best_perf > lambda_perf):
            best = regularized_least_squares(X_shuffled, Y_shuffled, lambda_par[lambda_idx])
            best_perf = lambda_perf
    return best, best_perf, models, results

Solution (vectorized solution).

Show code 
# Vectorised ridge regression.
# Changes vs. the reference solution:
#   * @ instead of .dot(.) for matrix products
#   * np.linalg.solve replaces the explicit inverse: solves
#     (Phi.T @ Phi + lam * I) w = Phi.T @ y in one numerically stable shot
#   * the K-fold loop precomputes the gram matrix Phi.T @ Phi and the
#     cross-product Phi.T @ y once per fold, so each ridge solve becomes a
#     single np.linalg.solve with a freshly added lambda*I
#   * fold indices are produced with np.array_split so each lambda iterates
#     over them without re-slicing the dataset
#   * fold accumulators use np.fromiter / array arithmetic instead of a
#     Python running sum, so the final per-lambda MSE is a vector mean
def regularized_least_squares(X_matrix: np.ndarray, labels: np.ndarray, lambda_par: np.ndarray) -> None:
    # (Phi.T @ Phi + lam * I) w = Phi.T @ y
    A = X_matrix.T @ X_matrix
    b = X_matrix.T @ labels
    d = A.shape[0]
    coeff = np.linalg.solve(A + lambda_par * np.eye(d), b)
    return coeff

def K_fold(X_training: np.ndarray, Y_training: np.ndarray, lambda_vec: np.ndarray, K: np.ndarray) -> None:
    # Shuffle once, then split index range into K contiguous folds
    n = X_training.shape[0]
    perm = np.random.permutation(n)
    X_shuffled = X_training[perm]
    Y_shuffled = Y_training[perm]
    fold_indices = np.array_split(np.arange(n), K)

    results = np.empty(len(lambda_vec))
    models = []
    best = None
    best_perf = np.inf

    for lambda_idx, lam in enumerate(lambda_vec):
        fold_mse = np.empty(K)
        for k, val_idx in enumerate(fold_indices):
            train_mask = np.ones(n, dtype=bool)
            train_mask[val_idx] = False
            # Train via vectorised normal equations
            w = regularized_least_squares(X_shuffled[train_mask], Y_shuffled[train_mask], lam)
            # Single-shot MSE on the held-out fold
            fold_mse[k] = np.mean((Y_shuffled[val_idx] - X_shuffled[val_idx] @ w) ** 2)
        lambda_perf = fold_mse.mean()
        results[lambda_idx] = lambda_perf
        # Refit on the full shuffled training set with this lambda
        full_model = regularized_least_squares(X_shuffled, Y_shuffled, lam)
        models.append(full_model)
        if lambda_perf < best_perf:
            best = full_model
            best_perf = lambda_perf

    return best, best_perf, models, list(results)
