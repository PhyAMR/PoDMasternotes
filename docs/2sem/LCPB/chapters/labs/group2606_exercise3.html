# 1. New dynamic build_model
def build_model(hp, clf, params_config):
    sampled_params = {}
    
    # 1. Create a dummy instance to see ALL supported parameters
    # This is much more reliable than inspect.signature for XGBoost
    try:
        temp_instance = clf()
        all_supported_params = temp_instance.get_params().keys()
    except Exception:
        # Fallback to signature if get_params fails
        all_supported_params = inspect.signature(clf).parameters.keys()

    # 2. Loop through our desired tuning config
    for p_name, p_settings in params_config.items():
        # Check if the model actually supports this parameter
        if p_name in all_supported_params:
            if p_settings['type'] == 'Int':
                sampled_params[p_name] = hp.Int(
                    p_name, p_settings['min'], p_settings['max'], step=p_settings.get('step', 1)
                )
            elif p_settings['type'] == 'Float':
                sampled_params[p_name] = hp.Float(
                    p_name, p_settings['min'], p_settings['max'], sampling=p_settings.get('sampling', 'linear')
                )
            elif p_settings['type'] == 'Categorical':
                sampled_params[p_name] = hp.Choice(
                    p_name, p_settings['values']
                )
    
    # This print will now show the full list!
    print(f"Building {clf.__name__} with: {list(sampled_params.keys())}")
    
    return clf(**sampled_params)

# 2. Define the "Master Config" for all possible parameters
tuning_config = {
    'objective' : {'type': 'Categorical', 'values': ['binary:logistic']},

import matplotlib.cm as cm
import matplotlib.colors as mcolors

label_dict = {
    'optimizer': 'Optimizer',
    'regularizer': 'Regularizer',
    'dropout': 'Dropout Rate',
    'learning_rate': 'Learning Rate'
}

def results_plot(tuner, num_trials=None, column="regularizer", row="optimizer", 
                 hue="dropout", x_var="learning_rate", label_dict=label_dict, 
                 sig_figs=3, title=None):
    
    if num_trials is None:
        num_trials = len(tuner.oracle.trials)
    trials = tuner.oracle.get_best_trials(num_trials=num_trials)

    if title is None:
        title = f"Parameter search for a model with {len(tuner.oracle.trials)} trials"

    results_list = []
    for trial in trials:
        data = trial.hyperparameters.values.copy()
        data['val_accuracy'] = round(trial.score, 3) if trial.score is not None else 0
        results_list.append(data)
    
    df = pd.DataFrame(results_list)

    # 1. Handling Missing Columns (Safety for different models)
    for col in [column, row, hue, x_var]:
        if col is not None and col not in df.columns:
            df[col] = "N/A"
    global_max = df['val_accuracy'].max()
    # 2. Rounding Logic
    def round_df_to_sig_figs(df_in, n_sigs):
        df_numeric = df_in.select_dtypes(include=[np.number])
        vals = df_numeric.values
        mask = (vals != 0) & (~np.isnan(vals))
        log_10 = np.floor(np.log10(np.abs(np.where(mask, vals, 1))))
        decimals = (n_sigs - 1) - log_10
        rounded = np.where(mask, np.round(vals * 10**decimals) / 10**decimals, vals)
        df_in[df_numeric.columns] = rounded
        return df_in

    df = round_df_to_sig_figs(df, sig_figs)

    # 3. FacetGrid Setup with Hue Logic
    colormap = None
    norm = None
    
    if hue is not None and hue in df.columns and df[hue].nunique() > 1:
        if df[hue].nunique() <= 9:
            # Discrete Case
            g = sns.FacetGrid(df, col=column, row=row, margin_titles=True, height=4, hue=hue, palette="Set2")
        else:
            # Continuous Case (Fixing the Palette/Hue swap)
            norm = mcolors.Normalize(vmin=df[hue].min(), vmax=df[hue].max())
            colormap = cm.get_cmap("cubehelix")
            unique_hues = sorted(df[hue].unique())
            palette_dict = {val: mcolors.to_hex(colormap(norm(val))) for val in unique_hues}
            
            # FIXED: hue is the COLUMN NAME, palette is the DICTIONARY
            g = sns.FacetGrid(df, col=column, row=row, margin_titles=True, height=4, hue=hue, palette=palette_dict)
    else:
        g = sns.FacetGrid(df, col=column, row=row, margin_titles=True, height=4)

    # 4. Annotate Function
    
    def annotate_scores(data, **kws):
        if data.empty:
            return
            
        ax = plt.gca()
        # 1. Identify which variable is on the y-axis
        # In your logic, this is either x_var or the 'groups_of' binned column
        y_var_name = (x_var if df[x_var].unique().shape[0] <= 10 else f"groups_of_{x_var}")
        
        # 2. Find the row with the maximum accuracy in THIS facet
        best_row_idx = data['val_accuracy'].argmax()
        max_score = data['val_accuracy'].iloc[best_row_idx]
        best_y_label = data[y_var_name].iloc[best_row_idx]
        
            
        # 3. Get the vertical position of that specific category
        # Matplotlib's categorical axes use integers 0, 1, 2... for positions
        
        # We find where our 'best_y_label' sits in the current y-ticks
        # A more robust way is using the category index from the plot
        y_pos = None
        ticks = ax.get_yticks()
        labels = [t.get_text() for t in ax.get_yticklabels()]
        
        try:
            # Match the label string to find the correct tick index
            label_idx = labels.index(str(best_y_label))
            y_pos = ticks[label_idx]
        except (ValueError, IndexError):
            # Fallback to middle if label matching fails
            y_pos = 0.5 
            transform = ax.transAxes
        else:
            # If we found the tick, we use Data coordinates for Y and Axes for X
            transform = ax.get_yaxis_transform() 

        # 4. Draw the text inside the bar or at the end of it
        if max_score < global_max:
            ax.text(0.95, y_pos, f"Score: {max_score:.3f}", 
                    transform=transform, 
                    fontsize=9, fontweight='bold',
                    ha='right', va='center', 
                    color=kws.get('color', 'black'),
                    bbox=dict(facecolor='white', alpha=0.8, edgecolor='none'))
        else:
            ax.text(0.95, y_pos, f"Score: {max_score:.3f}", 
                transform=transform, 
                fontsize=9, fontweight='bold',
                ha='right', va='center', 
                color=kws.get('color', 'black'),
                bbox=dict(facecolor='white', alpha=0.8, edgecolor='black', boxstyle='round,pad=0.2'))

    # 5. Plotting
    y_axis_var = x_var if x_var in df.columns else df.columns[0]
    
    if df[x_var].unique().shape[0] <= 10:  # Only annotate if there are 10 or fewer unique x values

        g.map_dataframe(
            sns.barplot,
            x="val_accuracy",
            y=y_axis_var,
            errorbar=None,
            orient="h"
            
        )
    else:
        df[f'groups_of_{x_var}'] = pd.cut(df[x_var], bins=10, duplicates='drop')
        g.map_dataframe(
            sns.barplot,
            x="val_accuracy",         # Numbers go on X
            y=f'groups_of_{x_var}',   # Categories go on Y
            errorbar=None,
            orient="h"                # Use "h" for clarity, or leave it to auto-detect
        )
    g.map_dataframe(annotate_scores)

    # 6. Final Layout and Margin Title Fixes
    # Push right margin to make room for titles
    g.fig.subplots_adjust(right=0.8, top=0.85, wspace=0.4) 
    g.fig.suptitle(title, fontsize=14, fontweight='bold')
    g.set_titles(col_template=f"{column}="+"{col_name}", row_template=f"{row}="+"{row_name}", fontweight='bold')

    # FIX: Reposition Row Titles to the far right
    for txt in g.fig.texts:
        if txt.get_rotation() == -90:
            txt.set_x(0.92) # Absolute figure coordinate
            txt.set_va('center')

    # Add Colorbar for Continuous Hue
    if colormap and norm:
        sm = plt.cm.ScalarMappable(cmap=colormap, norm=norm)
        sm.set_array([])
        cbar_ax = g.fig.add_axes([0.85, 0.15, 0.02, 0.6]) # [left, bottom, width, height]
        g.fig.colorbar(sm, cax=cbar_ax, label=label_dict.get(hue, hue))
    else:
        
        g.add_legend()
    g.set(xlim=(0, 1.1)) 
    plt.show()

def scat(ax, x, y, i=0, j=1, s=4, title="", xlabel=None, ylabel=None):
    """
    Scatter plot of data in the plane of features i and j.
    """
    # Use c=y to color points by their class label
    ax.scatter(x[:, i], x[:, j], s=s, c=y, cmap='viridis', alpha=0.7)
    ax.set_xlabel(xlabel if xlabel else f"Feature {i}")
    ax.set_ylabel(ylabel if ylabel else f"Feature {j}")
    ax.set_title(title)

def classify(clf, x_train, y_train, x_test, y_test, Lp, grid_range=50, dx=1, show=False, tree=False,name = "", params=""):
    """
    Trains a classifier, predicts on a grid to show decision boundaries, 
    and compares it to the training set.
    """
    # 1. Train and Evaluate
    clf.fit(x_train, y_train)
    accuracy = clf.score(x_test, y_test)
    error_rate = 100 * (1 - accuracy)
    
    print(f"Errors: {error_rate:.2f}% | Accuracy: {accuracy:.3f}")

    # 2. Create Artificial Grid for Decision Boundary Visualization
    # We create a 2D grid for the first two features
    s_points = np.arange(-grid_range, grid_range + dx, dx)
    xx, yy = np.meshgrid(s_points, s_points)
    
    # Flatten grid and prepare x_plot with Lp features (filling rest with zeros)
    grid_coords = np.c_[xx.ravel(), yy.ravel()]
    x_plot = np.zeros((grid_coords.shape[0], Lp))
    x_plot[:, :2] = grid_coords 
    
    print(f'Shape of x_plot: {x_plot.shape}')
    if show:
        # 3. Predict on the grid
        y_plot = clf.predict(x_plot)

        # 4. Visualization
        fig, AX = plt.subplots(1, 2, figsize=(10, 4))
        
        # Left: Decision Boundary (Predicted)
        scat(AX[0], x_plot, y_plot, i=0, j=1, s=dx*2, title="Decision Boundary (Predicted)")
        
        # Right: Actual Training Data
        scat(AX[1], x_train, y_train, i=0, j=1, s=10, title="Actual Training Set")
        
        fig.tight_layout()
        fig.suptitle(f"Results for model {name} with params \n {params}", y=1.2)
        plt.show()

    if tree:
        dump_list = clf.get_booster().get_dump() # get the list of trees in the model
        num_trees = len(dump_list)
        print("num_trees=",num_trees)

        fig, AX = plt.subplots(2,1,figsize=(12, 5))
        for i in range(min(2,num_trees)):
            ax=AX[i]
            plot_tree(clf, num_trees=i, ax=ax)
        fig.savefig("DATA/tree-classif.png", dpi=400, pad_inches=0.02)
        plt.show()

def plot_feature_importance(model, feature_names=None, title=None, ascending=False):
    """
    Plots feature importances for any scikit-learn or XGBoost model.
    - ascending: If True, smaller importances get darker colors.
    """
    importances = model.feature_importances_
    num_features = len(importances)
    
    # Use provided feature names or default to indices
    if feature_names is None:
        feature_names = [f"F{i}" for i in range(num_features)]
    
    # 1. Setup Colors using your Rescale logic
    my_cmap = plt.get_cmap("Reds")
    min_imp, max_imp = np.min(importances), np.max(importances)
    
    # Avoid division by zero if all importances are equal
    diff = (max_imp - min_imp) if max_imp != min_imp else 1
    
    if not ascending:
        # Standard: Higher importance = Darker Red
        rescale = lambda y: 0.3 + 0.7 * (y - min_imp) / diff
    else:
        # Reversed: Lower importance = Darker Red
        rescale = lambda y: 0.3 + 0.7 * (max_imp - y) / diff

    # 2. Create Plot
    plt.figure(figsize=(8, 4))
    bars = plt.bar(range(num_features), importances, 
                   color=my_cmap(rescale(importances)),
                   edgecolor='black', linewidth=0.5)
    
    # 3. Styling
    plt.xlabel("Features")
    plt.ylabel("Importance Score")
    
    # Determine the type of importance used (XGBoost specific)
    imp_type = getattr(model, 'importance_type', 'Gini/MDI')
    plt.title(title if title else f"Feature Importance ({imp_type})")
    
    plt.xticks(range(num_features), feature_names, rotation=45)
    plt.grid(axis='y', linestyle='--', alpha=0.7)
    
    plt.tight_layout()
    plt.show()


def plot_xgb_multi_importance(model, feature_names=None):
    """
    Compares different XGBoost importance types. 
    Handles cases where booster.feature_names is None.
    """
    list_type = ["weight", "gain", "total_gain", "cover", "total_cover"]
    list_col = ["#7565F0", "k", "#666666", "#F0A000", "gold"]
    
    booster = model.get_booster()
    
    # FIX: If booster.feature_names is None, generate generic names
    # based on the model's expected input shape
    if booster.feature_names is not None:
        all_features = booster.feature_names
    else:
        # Fallback: get count from the actual feature_importances_ array
        num_features = len(model.feature_importances_)
        all_features = [f"f{i}" for i in range(num_features)]
    
    L = len(all_features)
    
    # Use the names passed in the function call for the X-axis labels
    if feature_names is None:
        display_names = all_features
    else:
        display_names = feature_names

    plt.figure(figsize=(12, 6))

    for i, t in enumerate(list_type):
        score_dict = booster.get_score(importance_type=t)
        
        # XGBoost get_score keys are strings like 'f0', 'f1' regardless of names
        # so we map the scores carefully
        values = np.array([score_dict.get(f, 0) for f in all_features])
        
        total = np.sum(values)
        if total > 0:
            values = values / total
        
        width = 0.12
        offset = (i - len(list_type)/2) * width + (width/2)
        
        plt.bar(np.arange(L) + offset, values, 
                color=list_col[i], width=width, label=t, alpha=0.8)

    plt.xlabel("Features", fontweight='bold')
    plt.ylabel("Importance (Normalized to 1)", fontweight='bold')
    plt.title("XGBoost: Comparison of Importance Metrics", fontsize=14)
    
    # Use our display_names for the ticks
    plt.xticks(np.arange(L), display_names, rotation=45)
    plt.legend(title="Importance Type", bbox_to_anchor=(1.05, 1), loc='upper left')
    plt.grid(axis='y', linestyle='--', alpha=0.3)
    
    plt.tight_layout()
    plt.show()

from sklearn.model_selection import KFold
k=5
fac = np.linspace(0.1, 0.95, 20)

results = {}
for (n, tunner),( _, clf) in zip(tunners.items(),clasifiers.items()):
    
    
    
    accuracies_fnn = []
    losses_fnn = []
    for f in fac:

        total_acc = 0
        total_loss = 0
        x_train, y_train, x_test, y_test, N, L = load_data("XGB_26", fac=f)
        for train_index, test_index in KFold(n_splits=k, shuffle=True, random_state=1234).split(x_train):

            total_acc += accuracy
            total_loss += error_rate

        avg_loss = total_loss/ k

# 1. New dynamic build_model
def build_model(hp, clf, params_config):
    sampled_params = {}
    
    # 1. Create a dummy instance to see ALL supported parameters
    # This is much more reliable than inspect.signature for XGBoost
    try:
        temp_instance = clf()
        all_supported_params = temp_instance.get_params().keys()
    except Exception:
        # Fallback to signature if get_params fails
        all_supported_params = inspect.signature(clf).parameters.keys()

    # 2. Loop through our desired tuning config
    for p_name, p_settings in params_config.items():
        # Check if the model actually supports this parameter
        if p_name in all_supported_params:
            if p_settings['type'] == 'Int':
                sampled_params[p_name] = hp.Int(
                    p_name, p_settings['min'], p_settings['max'], step=p_settings.get('step', 1)
                )
            elif p_settings['type'] == 'Float':
                sampled_params[p_name] = hp.Float(
                    p_name, p_settings['min'], p_settings['max'], sampling=p_settings.get('sampling', 'linear')
                )
            elif p_settings['type'] == 'Categorical':
                sampled_params[p_name] = hp.Choice(
                    p_name, p_settings['values']
                )
    
    # This print will now show the full list!
    print(f"Building {clf.__name__} with: {list(sampled_params.keys())}")
    
    return clf(**sampled_params)

# 2. Define the "Master Config" for all possible parameters
tuning_config = {
    'objective' : {'type': 'Categorical', 'values': ['binary:logistic']},
    'importance_type' : {'type': 'Categorical', 'values': ['gain']},
    'n_estimators': {'type': 'Int', 'min': 50, 'max': 300, 'step': 50},
    'max_depth': {'type': 'Int', 'min': 3, 'max': 10},
    'learning_rate': {'type': 'Float', 'min': 1e-5, 'max': 1e-3, 'sampling': 'log'},
    'reg_lambda': {'type': 'Float', 'min': 1e-3, 'max': 1e-1, 'sampling': 'log'}
    

}

# 3. The Optimized Loop
n_trials = 200
tunners = {}
clasifiers = {
    "XGBClassifier": XGBClassifier,
    "GradientBoostingClassifier": GradientBoostingClassifier,
    "AdaBoostClassifier": AdaBoostClassifier,
}

for name, clf in clasifiers.items():

    tuner = kt.tuners.SklearnTuner(

            objective=kt.Objective('score', 'max'),
            max_trials=n_trials
        ),
        # Pass the classifier and the full config; the function will filter them
        hypermodel=lambda hp, c=clf: build_model(hp, c, tuning_config),
        scoring=metrics.make_scorer(metrics.accuracy_score),
        cv=model_selection.StratifiedKFold(5),
        directory='ParameterTuning',
        project_name=name,
        overwrite=False
    )

def results_plot(tuner, num_trials=None, column="regularizer", row="optimizer", 
                 hue="dropout", x_var="learning_rate", label_dict=label_dict, 
                 sig_figs=3, title=None):

        num_trials = len(tuner.oracle.trials)
    trials = tuner.oracle.get_best_trials(num_trials=num_trials)

    if title is None:
        title = f"Parameter search for a model with {len(tuner.oracle.trials)} trials"

    results_list = []
    for trial in trials:
        data = trial.hyperparameters.values.copy()
        data['val_accuracy'] = round(trial.score, 3) if trial.score is not None else 0
        results_list.append(data)
    
    df = pd.DataFrame(results_list)

    # 1. Handling Missing Columns (Safety for different models)
    for col in [column, row, hue, x_var]:
        if col is not None and col not in df.columns:
            df[col] = "N/A"

    def round_df_to_sig_figs(df_in, n_sigs):
        df_numeric = df_in.select_dtypes(include=[np.number])
        vals = df_numeric.values
        mask = (vals != 0) & (~np.isnan(vals))
        log_10 = np.floor(np.log10(np.abs(np.where(mask, vals, 1))))
        decimals = (n_sigs - 1) - log_10
        rounded = np.where(mask, np.round(vals * 10**decimals) / 10**decimals, vals)
        df_in[df_numeric.columns] = rounded
        return df_in

    df = round_df_to_sig_figs(df, sig_figs)

    colormap = None

    else:

    # 4. Annotate Function
    
    def annotate_scores(data, **kws):
        if data.empty:

        ax = plt.gca()

        y_var_name = (x_var if df[x_var].unique().shape[0] <= 10 else f"groups_of_{x_var}")

def plot_feature_importance(model, feature_names=None, title=None, ascending=False):
    """
    Plots feature importances for any scikit-learn or XGBoost model.
    - ascending: If True, smaller importances get darker colors.
    """
    importances = model.feature_importances_
    num_features = len(importances)
    
    # Use provided feature names or default to indices
    if feature_names is None:
        feature_names = [f"F{i}" for i in range(num_features)]
    
    # 1. Setup Colors using your Rescale logic
    my_cmap = plt.get_cmap("Reds")
    min_imp, max_imp = np.min(importances), np.max(importances)
    
    # Avoid division by zero if all importances are equal
    diff = (max_imp - min_imp) if max_imp != min_imp else 1
    
    if not ascending:
        # Standard: Higher importance = Darker Red
        rescale = lambda y: 0.3 + 0.7 * (y - min_imp) / diff
    else:
        # Reversed: Lower importance = Darker Red
        rescale = lambda y: 0.3 + 0.7 * (max_imp - y) / diff

    # 2. Create Plot
    plt.figure(figsize=(8, 4))
    bars = plt.bar(range(num_features), importances, 
                   color=my_cmap(rescale(importances)),
                   edgecolor='black', linewidth=0.5)
    
    # 3. Styling
    plt.xlabel("Features")
    plt.ylabel("Importance Score")
    
    # Determine the type of importance used (XGBoost specific)
    imp_type = getattr(model, 'importance_type', 'Gini/MDI')
    plt.title(title if title else f"Feature Importance ({imp_type})")
    
    plt.xticks(range(num_features), feature_names, rotation=45)
    plt.grid(axis='y', linestyle='--', alpha=0.7)
    
    plt.tight_layout()
    plt.show()

from keras.models import Sequential
from keras.layers import Dense, Dropout

def FFNN(x_train,x_test,y_train,y_test,epochs=200):
    model = Sequential()
    model.add(Dense(L,input_shape=(L,),activation='relu'))
    model.add(Dense(20,activation='relu'))
    model.add(Dropout(0.1)) #Helps with overfitting, it prevents 
    model.add(Dense(20,activation='relu'))
    model.add(Dropout(0.1))
    model.add(Dense(20,activation='relu'))
    model.add(Dropout(0.1))
    model.add(Dense(1,activation='sigmoid'))
    
    model.compile(loss='binary_crossentropy',
              optimizer='RMSprop',
              #optimizer='adam',
              metrics=['accuracy'])
    model.fit(x_train, y_train,
               epochs = epochs, batch_size = 50,
               validation_data=(x_test,y_test),
               verbose=2)
    accuracy, loss = model.evaluate(x_test, y_test, verbose=0)

    return accuracy, loss
